\subsection{\textsf{Schur's Inequality}}
Let $a$,$b$,$c$ be nonnegative real numbers and $r>0$ be positive real
number. Then
\[
	\cyc a^r(a^2+bc) \ge \cyc a^{r+1}(b+c).
\]
Equality occurs if $a=b=c$ or $a=b$ and $c=0$ and permutations.

For example, when $r=1$:
\[
	\cyc (a^3+abc) \ge \cyc a^2b.
\]

An easier way to write the Schur's Inequality is the following:
\[
	\cyc a^r(a-b)(a-c) \ge 0.
\]
By the way, using this form of the Schur's Inequality, we can even prove it.
Say WLOG, that $a \ge b \ge c$. It can be done because it is a symmetrical
sum over $a$,$b$,$c$. Therefore, $(a-b)$ is positive, as is $(a-c)$. \qed

\begin{example}{}{}
	Prove
	\[
		a^2+b^2+c^2+(abc)^{\frac{1}{3}}\sqrt{\frac{ab+bc+ca}{3}} \ge \frac{4}
		{3}(ab+bc+ca).
	\]
\end{example}

\textbf{\textcolor{color2}{Solution.}}

\begin{align*}
	a^2+b^2+c^2+(abc)^{\frac{1}{3}}\sqrt{\frac{ab+bc+ca}{3}} & \ge \frac{a^2+b^2+c^2}{3} + (abc)^{\frac{2}{3}} \\ & \ge
	\frac{1}{3} \sym a^{\frac{4}{3}}b^{\frac{2}{3}}                                                            \\ & \ge
	\frac{2}{3} \cyc ab.
\end{align*}
Moreover,
\[
	\frac{2}{3}(a^2+b^2+c^2) + \frac{2}{3}(ab+bc+ca) \ge \frac{2}{3}(ab+bc+ca) + \frac{2}{3}(ab+bc+ca)
\]

\begin{insight}
	The hardest and most important step here is dividing $a^2+b^2+c^2$ by $3$, and
	later, using the Schur's Inequality. In order to notice it, you needed to see
	that $(abc)^{\frac{1}{3}}$ by itself is weak and doesn't help much. Therefore,
	using a fraction of $a^2+b^2+c^2$ to ``complete" it, would be really helpful.
	However, the Schur step is harder to notice without a previous pattern
	recognition. Nevertheless, it is still possible to see that
	$a^2=(a^{\frac{2}{3}})^3$ and $a^{\frac{4}{3}}b^{\frac{2}{3}} = (a^2b)^{\frac{2}{3}}$
\end{insight}

\subsection{\textsf{Weighted Power Mean}}

Let $a_1, \dots, a_n$ and $w_1,\dots,w_n$ be positive real numbers, where $w_1+\dots+w_n=1$.
Besides that, let $r$ be any real number. Therefore,

\[
	P(r) =
	\begin{cases}
		(w_1a^r_1+\dots+w_na_n^r)^{\frac{1}{r}} \iff r \neq 0 \\
		a_1^{w_1}\dots a_n^{w_n} \iff r =0.
	\end{cases}
\]

If $s>r$, $P(s)>P(r)$.

\begin{example}{\textcolor{teal}{Taiwan TST 2014}}{}
	Let $a$,$b$,$c >0$. Prove that
	\[
		3(a+b+c) \ge 8\sqrt[3]{abc} + \sqrt[3]{\frac{a^3+b^3+c^3}{3}}
	\]
\end{example}
\textbf{\textcolor{color2}{Solution.}} First, divide everything by $9$ and raise everything to
the power of $3$. Therefore, we gotta prove
\[
	\left(\frac{1}{3}(a+b+c)\right)^3 \ge \left(\frac{8}{9}(abc)^{\frac{1}{3}}+\frac{1}{9}\left(\frac{a^3+b^3+c^3}{3}\right)^
	{\frac{1}{3}}\right)^3.
\]
However, by the Weighted Power Mean,

\[
	\left(\frac{8}{9}(abc)^{\frac{1}{3}}+\frac{1}{9}\left(\frac{a^3+b^3+c^3}{3}\right)^{\frac{1}{3}}\right)^3 \le
	\frac{8}{9}(abc)+\frac{1}{9}\left(\frac{a^3+b^3+c^3}{3}\right).
\]
Hence, we must prove that

\[
	\left(\frac{(a+b+c)}{3}\right)^3 \ge \frac{24abc+a^3+b^3+c^3}{27} \iff (a+b+c)^3 \ge 24abc+a^3+b^3+c^3. \qed
\]

\subsubsection{All Russian Olympiad 2018 Grade 11 P2}
\begin{problem}
Let $n\geq 2$ and $x_{1},x_{2},\ldots,x_{n}$ positive real numbers. Prove that
\[
	\frac{1+x_{1}^2}{1+x_{1}x_{2}}+\frac{1+x_{2}^2}{1+x_{2}x_{3}}+\cdots+\frac{1+x_{n}^2}{1+x_{n}x_{1}}\geq n.
\]
\end{problem}

By the AM-GM inequality,
\[
	\cyc \frac{1+x^2_i}{1+x_ix_{i+1}} \ge
	n\sqrt[n]{\prodcyc \frac{1+x^2_i}{1+x_ix_{i+1}}}.
\]
Therefore, it suffices to show that
\[
	\prodcyc \frac{1+x^2_i}{1+x_ix_{i+1}} \ge
	1 \iff \prodcyc (1+x^2_i) \ge \prodcyc (1+x_ix_{i+1}),
\]
which is clearly true, since
\[
	(1+x^2_i)(1+x^2_{i+1}) \ge (1+x_ix_{i+1})^2
\]
by the Cauchy-Schwarz inequality. Hence,
\[
	\prodcyc (1+x^2_i) \ge \prodcyc (1+x_1x_{i+1}).
\]
Thus,
\[
	\cyc \frac{1+x^2_i}{1+x_ix_{i+1}} \ge n.
\]

\subsubsection{IMO 2000 P2}
\begin{problem}
Let $ a, b, c$ be positive real numbers so that $ abc = 1$. Prove that
\[
	\left( a - 1 + \frac 1b \right) \left( b - 1 + \frac 1c \right) \left( c - 1 + \frac 1a \right) \leq 1.
\]
\end{problem}

\textbf{\textcolor{color2}{First solution}} Since $abc=1$, let $a=\frac{x}{y}$, $b=\frac{y}{z}$ and $c=\frac{z}{x}$. Therefore, it suffices to prove that
\[
	\prodcyc \left(\frac{x}{y}-1+\frac{z}{y}\right) = \prodcyc \left(\frac{x-y+z}{y}\right) \le 1 \iff
	\prodcyc \left(x-y+z\right) \le xyz.
\]
Let $u=x-y+z$,$v=y-z+x$ and $w=z-x+y$. Hence, it is enough to show that
\[
	uvw \le \frac{(u+v)(v+w)(w+u)}{8},
\]
which is true by AM-GM. Thus,
\[
	\prodcyc \left( a - 1 + \frac 1b \right) \le 1.
\]

\textbf{\textcolor{color2}{Second solution}} Since $abc=1$, let $a=\frac{x}{y}$, $b=\frac{y}{z}$ and $c=\frac{z}{x}$. Therefore, it suffices to prove that
\[
	\prodcyc \left(\frac{x}{y}-1+\frac{z}{y}\right) = \prodcyc \left(\frac{x-y+z}{y}\right) \le 1 \iff
	\prodcyc \left(x-y+z\right) \le xyz.
\]
Now, notice that
\[
	\prodcyc \left(x-y+z\right) = \sym x^2y -2xyz -x^3-y^3-z^3.
\]
Hence, it is enough to show that
\[
	\sym x^2y -x^3-y^3-z^3 \le 3xyz,
\]
which is true by the Schur's inequality. Thus,
\[
	\prodcyc \left( a - 1 + \frac 1b \right) \le 1.
\]

\begin{insight}
	$a=\frac{x}{y}$, $b=\frac{y}{z}$ and $c=\frac{z}{x}$ is the best possible strategy when $abc=1$, always remember that.

	Both the Schur's and the AM-GM methods didn't come to my mind, even though they were really easy. I feel that the
	Schur's method didn't come to my mind only because I was lazy, because it was so easy. However, the AM-GM method was
	harder. I have been noticing a pattern that follows like this: an equality must obligatorily be true in order for the
	question to be true. Hence, if it easy, basically every ineq you know should be tested, just because those inequalities
	seems to be easy.
\end{insight}

\begin{problem}
Let $x,y$ and $z$ be positive real numbers such that $xyz=1$. Prove that
\[
	\frac{x^{3}}{(1 + y)(1 + z)}+\frac{y^{3}}{(1 + z)(1 + x)}+\frac{z^{3}}{(1 + x)(1 + y)}
	\geq \frac{3}{4}.
\]
\end{problem}

\textcolor{color1}{\textbf{First solution}} By Cauchy-Schwarz,

\begin{align*}
	\left(\cyc \frac{x^3}{(1+y)(1+z)}\right)\left(\cyc (1+y)(1+z)\right) & \geq (x^{\frac{3}{2}}
	+ y^{\frac{3}{2}} + z^{\frac{3}{2}})^2                                                       \\ & = x^3+y^3+z^3+2((xy)^{\frac{3}{2}}+(yz)^{\frac{3}{2}}+(zx)^{\frac{3}{2}}).
\end{align*}
Therefore, it suffices to prove that
\begin{align*}
	 & x^3+y^3+z^3+2((xy)^{\frac{3}{2}}+(yz)^{\frac{3}{2}}+(zx)^{\frac{3}{2}}) \geq \frac{3}{4}\left(3+2(x+y+z)+xy+yz+zx \right) \iff            \\
	 & 2 \sym x^3 +4 \sym (xy)^{\frac{3}{2}} \geq \frac{3}{2} \sym xyz + 3 \sym(x^5y^2z^2)^{\frac{1}{3}}+\frac{3}{2} \cyc(x^4y^4z)^{\frac{1}{3}}
\end{align*}

\textcolor{color1}{\textbf{Second solution}} By Titu's lemma,

\[
	\cyc \frac{x^4}{x(1+y)(1+z)} \geq \frac{(x^2+y^2+z^2)^2}{3xyz+2(xy+yz+za)+x+y+z} \geq \frac{(x^2+y^2+z^2)^2}{4(x^2+y^2+z^2)} \geq \frac{3}{4}
\]

\begin{problem}
Let $ a$, $ b$, $ c$ be positive real numbers such that $ abc = 1$. Prove that
\[
	\frac {1}{a^{3}\left(b + c\right)} + \frac {1}{b^{3}\left(c + a\right)} + \frac {1}{c^{3}\left(a + b\right)}\geq \frac {3}{2}.
\]
\end{problem}

\textcolor{color1}{\textbf{First solution}} By Cauchy-Schwarz,
\[
	\left(\cyc \frac{1}{a^3(b+c)}\right)\left(\cyc a(b+c)\right) \geq
	\left(\frac{1}{a} + \frac{1}{b} + \frac{1}{c}\right)^2 = (ab+bc+ca)^2.

\]
which is true by AM-GM.

\textcolor{color1}{\textbf{Second solution}} By Cauchy-Schwarz,
\[
	\left(\cyc \frac{1}{a^3(b+c)}\right)\left(\cyc a(b+c)\right) \geq
	\left(\frac{1}{a} + \frac{1}{b} + \frac{1}{c}\right)^2 = (ab+bc+ca)^2,
\]
and by Titu's lemma,
\[
	\cyc \frac{(\frac{1}{a})^2}{a(b+c)} \geq \frac{ab+bc+ca}{2} \ge \frac{3(abc)^{\frac{2}{3}}}{2} = \frac{3}{2}
\]

\subsubsection{IMO Shortlist 2004 A5}
\begin{problem}
If $a$, $b$ ,$c$ are three positive real numbers such that $ab+bc+ca = 1$, prove that\[ \sqrt[3]{ \frac{1}{a} + 6b} + \sqrt[3]{\frac{1}{b} + 6c} + \sqrt[3]{\frac{1}{c} + 6a } \leq \frac{1}{abc}.  \]
\end{problem}

\textbf{\textcolor{color2}{First solution}} After homogenizing the inequality, we must prove that

\begin{align*}
	 & \cyc \sqrt[3]{\frac{(ab+bc+ca)^2}{a}+6b(ab+bc+ca)} & =   \\
	 & \cyc \sqrt[3]{\frac{(ab+bc+ca)(7ab+bc+ca)}{a}}     & \le \\
	 & \frac{(ab+bc+ca)^2}{abc}.
\end{align*}
By the HÃ¶lder's inequality,
\begin{align*}
	\left(\cyc \sqrt[3]{\frac{(ab+bc+ca)(7ab+bc+ca)}{a}}\right)^3 \le \left(\cyc ab+bc+ca\right)\left(\cyc 7ab+bc+ca\right)\left(\cyc \frac{1}{a}\right).
\end{align*}
Therefore, it suffices to prove the following.
\begin{align*}
	\left(\cyc ab+bc+ca\right)\left(\cyc 7ab+bc+ca\right)\left(\cyc \frac{1}{a}\right) & \le \frac{(ab+bc+ca)^6}{(abc)^3}. & \iff \\
	\frac{24(ab+bc+ca)^3}{abc}                                                         & \le \frac{(ab+bc+ca)^6}{(abc)^3}  & \iff \\
	24(abc)^2                                                                          & \le (ab+bc+ca)^3                  & \iff \\
	2\sqrt[3]{3}(abc)^{\frac{2}{3}}                                                    & \le ab+bc+ca.
\end{align*}
By the AM-GM inequality, $ab+bc+ca \ge 3(abc)^{\frac{2}{3}}$. Hence, we must prove that $3 \ge 2\sqrt[3]{3}$.
Which is true, since $27>24$.

Consequently,
\[
	\frac{1}{abc} \ge \cyc \sqrt[3]{\frac{1}{a} +6b}.
\]

\textbf{\textcolor{color2}{Second solution, \href{https://artofproblemsolving.com/community/c6h30865p23434282}{found by DottedCalculator}}} By the HÃ¶lder's inequality,
\begin{align*}
	\left(\cyc \left(\frac{1}{a}+6b\right)\right)\left(\cyc 1\right)^2 & \ge \left(\cyc \sqrt[3]{\frac{1}{a}+6b}\right)^3.
	% \sqrt[3]{54(a+b+c)+9\left(\frac{1}{a}+\frac{1}{b}+\frac{1}{c}\right)} & \ge\cyc \sqrt[3]{\frac{1}{a}+6b}.
\end{align*}
Therefore, it suffices to prove that
\begin{align*}
	\frac{1}{(abc)^3} \ge 54(a+b+c)+9\left(\frac{1}{a}+\frac{1}{b}+\frac{1}{c}\right) \iff
	1 \ge 54(abc)^3(a+b+c) + 9(abc)^2 = \\
	9(abc)^2(6abc(a+b+c)+1)
\end{align*}
since $ab+bc+ca=1$.

By the AM-GM inequality, $\frac{1}{27} \ge (abc)^2$, and by the Titu's Lemma, $a^2b^2+b^2c^2+c^2a^2 \ge \frac{(ab+bc+ca)^2}{3} = \frac{1}{3}$.
Hence, $9(abc)^2 \le \frac{1}{3}$ and $6abc(a+b+c)+1 \le 3$. It happens because
\begin{align*}
	6abc(a+b+c)+1 = 3(ab+bc+ca)^2 -3(a^2b^2+b^2c^2+c^2a^2)+1=4-3(a^2b^2+b^2c^2+c^2a^2).
\end{align*}
Thus,
\begin{align*}
	1 \ge 9(abc)^2(6abc(a+b+c)+1).
\end{align*}
Consequently,
\begin{align*}
	\frac{1}{abc} \ge \cyc \sqrt[3]{\frac{1}{a} +6b}.
\end{align*}

\subsubsection{IMO 2004 P4}

\begin{problem}
Let $n \geq 3$ be an integer. Let $t_1$, $t_2$, ..., $t_n$ be positive real numbers such that\[n^2 + 1 > \left( t_1 + t_2 + \cdots + t_n \right) \left( \frac{1}{t_1} + \frac{1}{t_2} + \cdots + \frac{1}{t_n} \right).\]Show that $t_i$, $t_j$, $t_k$ are side lengths of a triangle for all $i$, $j$, $k$ with $1 \leq i < j < k \leq n$.
\end{problem}
\textbf{\textcolor{color1}{Solution.}} \quad First, notice that
\[
	n^2+1>(t_1+\dots+t_n)\left(\frac{1}{t_1}+\dots+\frac{1}{t_n}\right) = n+ \cyc \left(\frac{t_r}{t_s}+\frac{t_s}{t_r}\right)
\]
for all $1\le r < s \le n$. At the same time, by the AM-GM inequality,
\[
	\frac{t_r}{t_s}+\frac{t_s}{t_r}\ge2 \quad \text{ and } \quad n+2\binom{n}{2}=n^2.
\]
Hence,
\[
	n+ 2\left(\binom{n}{2}-3\right) + \left(\frac{t_i}{t_j}+\frac{t_j}{t_i}\right) + \left(\frac{t_j}{t_k}+\frac{t_k}{t_j}\right) + \left(\frac{t_k}{t_i}+\frac{t_i}{t_k}\right) \ge n^2.
\]
Consequently,
\[
	\left(\frac{t_i}{t_j}+\frac{t_j}{t_i}\right) + \left(\frac{t_j}{t_k}+\frac{t_k}{t_j}\right) + \left(\frac{t_k}{t_i}+\frac{t_i}{t_k}\right)
	< 7 \iff \left(\frac{t_j}{t_k}+\frac{t_k}{t_j}\right) + \left(\frac{t_k}{t_i}+\frac{t_i}{t_k}\right) = \frac{t_i+t_j}{t_k}+t_k\left(\frac{1}{t_j}+\frac{1}{t_i}\right) < 5
\]
by the AM-GM inequality. Moreover, by the Cauchy-Schwarz inequality,
\[
	\left(\frac{1}{t_j}+\frac{1}{t_i}\right)\left(t_j+t_i\right) \ge 4 \iff \frac{1}{t_j}+\frac{1}{t_i} \ge \frac{4}{t_j+t_i}.
\]
Thus,
\[
	\frac{t_i+t_j}{t_k}+\frac{4t_k}{t_j+t_i} < 5 \iff \left(\frac{t_i+t_j}{t_k}\right)^2-5\left(\frac{t_i+t_j}{t_k}\right)+4 = \left(\frac{t_i+t_j}{t_k}-1\right)\left(\frac{t_i+t_j}{t_k}-4\right) < 0,
\]
which occurs if and only if $1<\frac{t_j+t_i}{t_k}<4$. However, if $\frac{t_j+t_i}{t_k}>1$, $t_k<t_j+t_i$.

Therefore, $t_i$, $t_j$, $t_k$ are side lengths of a triangle.

\subsubsection{IMO 2012 P2}

\begin{problem}
Let $n\ge 3$ be an integer, and let $a_2,a_3,\ldots ,a_n$ be positive real numbers such that $a_{2}a_{3}\cdots a_{n}=1$. Prove that
\[(1 + a_2)^2 (1 + a_3)^3 \dotsm (1 + a_n)^n > n^n.\]
\end{problem}

Notice
\[
	a_k+1=a_k+\frac{1}{k-1}+\cdots+\frac{1}{k-1} \ge k\sqrt[k]{\frac{a_k}{(k-1)^{k-1}}} \implies (1+a_k)^k = \frac{k^k}{(k-1)^{k-1}}a_k \quad \forall k \ge 2
\]
by AM-GM. Hence,
\[
	(1 + a_2)^2 (1 + a_3)^3 \dotsm (1 + a_n)^n \ge \prod_{k=2}^{n} \frac{k^k}{(k-1)^{k-1}} \cdot a_2a_3\cdots a_n =
	n^n.
\]
Equality holds if and only if $a_k=\frac{1}{k-1}$, which is impossible for $n \ge 3$.
Thus,
\[
	(1 + a_2)^2 (1 + a_3)^3 \dotsm (1 + a_n)^n > n^n.
\]

\begin{insight}
	$a_k+1=a_k+(k-1)\frac{1}{k-1}$.
\end{insight}

\[
	9\sqrt[3]{abc} \ge 3(\sqrt{ab}+\sqrt{bc}+\sqrt{ca}),
\]
and by QM-AM,
\[
	\cyc \left(\sqrt{\frac{a^3+b^3}{a+b}}+3\sqrt{ab}\right) \le \cyc 4\sqrt{\frac{a^2-ab+b^2+3ab}{4}} = \cyc 2(a+b) = 12.
\]
Hence,
\[\sum_{\text{cyc}}\sqrt{\frac{a^3+b^3}{a+b}}+9\sqrt[3]{abc}\le12.\]

\textbf{\textcolor{color1}{Second solution}} By Cauchy-Schwarz,
\begin{align*}
	 & \left(\cyc(\sqrt{a^2-ab+b^2} + 3\sqrt{ab})\right)^2 &  & \le \cyc(a^2-ab+b^2 + 3ab)(1+3) \implies \\
	 & \cyc(\sqrt{a^2-ab+b^2} + 3\sqrt{ab})                &  & \le 12.
\end{align*}
However, $3(\sqrt{ab}+\sqrt{bc}+\sqrt{ca}) \le 9\sqrt[3]{abc}$. Hence,
\[\sum_{\text{cyc}}\sqrt{\frac{a^3+b^3}{a+b}}+9\sqrt[3]{abc}\le12.\]
